import pygambit as gbt

# Create the game with two players
g = gbt.Game.new_tree(players=["Player 1", "Player 2"],
                      title="Six-chamber alternating Russian Roulette")

# Reasoning (in comments):
# Step 1: Public state is (remaining chambers, whose turn). There is no imperfect information
# between different remaining-chamber states, so no infosets for players are needed.
# Step 2: We build the game tree explicitly without loops or recursion.
# Step 3: At each decision node the acting player has two actions: "Quit" or "Pull".
# Step 4: If "Quit" -> immediate terminal: quitting player gets 0, other gets 1.
# Step 5: If "Pull" and remaining > 1 -> chance node with probabilities:
#         fire with probability 1/remaining, click with probability (remaining-1)/remaining.
#         If fire -> acting player dies (payoff -1 to actor, +1 to other).
#         If click -> remaining reduces by 1 and turn passes to other player.
# Step 6: If "Pull" and remaining == 1 -> pull fires for certain (no chance node),
#         acting player dies immediately.

# Add commonly used outcomes to reuse across terminals
out_p1_quit = g.add_outcome([0, 1], label="P1 quits")    # P1 withdraws, P2 wins
out_p2_quit = g.add_outcome([1, 0], label="P2 quits")    # P2 withdraws, P1 wins
out_p1_shot = g.add_outcome([-1, 1], label="P1 shot")    # P1 shoots himself
out_p2_shot = g.add_outcome([1, -1], label="P2 shot")    # P2 shoots himself

# -------- Move 1: Player 1 with 6 chambers remaining --------
# Player 1 chooses Quit or Pull
g.append_move(g.root, "Player 1", ["Quit", "Pull"])
# If P1 quits immediately, he gets 0 and P2 gets 1
g.set_outcome(g.root.children[0], out_p1_quit)
