# - If they Pull and there are n>1 chambers remaining, chance fires with probability 1/n and the player dies,
#   otherwise (probability (n-1)/n) play continues with one fewer unpulled chamber and the other player's turn.

# - We'll explicitly build the game tree without loops or recursion, unrolling all possible histories

g = gbt.Game.new_tree(players=["Player 1", "Player 2"],

g.append_move(g.root, "Player 1", ["Quit", "Pull"])
# Outcomes we'll reuse
p1_dies = g.add_outcome([-1, 1], label="P1 dies")

p2_quits = g.add_outcome([1, 0], label="P2 quits")
