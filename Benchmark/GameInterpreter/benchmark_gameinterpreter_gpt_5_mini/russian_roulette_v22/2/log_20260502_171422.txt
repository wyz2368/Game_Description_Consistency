import pygambit as gbt

# Two-player alternating Russian roulette with a six-chamber revolver.
# Players: "Player 1" (acts first) and "Player 2".
# Payoffs:
# - If a player shoots himself: shooter gets -1, other gets 1.
# - If a player quits: quitter gets 0, other gets 1.
#
# Reasoning (encoded as comments below, step-by-step):
# - The public history (all past pulls and their results) is common knowledge,
#   so this is a perfect-information game: no information sets need merging.
# - Start with 6 unpulled chambers. Each successful "no-fire" observation
#   reduces the remaining chambers by one and play passes to the other player.
# - For a pull when k > 1 chambers remain, chance determines "Fire" (probability 1/k)
#   or "NoFire" (probability (k-1)/k). When k == 1, a pull fires for certain.
# - We explicitly construct the tree for the maximum of 6 pulls (no loops).
#
# Implementation notes:
# - Avoid loops and recursion: every node is constructed explicitly.
# - Use gbt.Rational for chance probabilities.
# - Do not include chance branches with probability zero (we never do).
# - We define outcome objects and reuse them where appropriate.

g = gbt.Game.new_tree(players=["Player 1", "Player 2"],
                      title="Six-chamber alternating Russian roulette")

# Predefine outcome objects to reuse
# Shooter dies -> shooter -1, other +1
p1_shot = g.add_outcome([-1, 1], label="P1 shot (P2 wins)")
p2_shot = g.add_outcome([1, -1], label="P2 shot (P1 wins)")
# Quit outcomes: quitter 0, other 1
p1_quits = g.add_outcome([0, 1], label="P1 quits (P2 wins)")
p2_quits = g.add_outcome([1, 0], label="P2 quits (P1 wins)")

# Root decision: Player 1 with 6 chambers remaining.
# Actions: "Quit" (immediate terminal), "Pull" (chance when >1 chamber)
g.append_move(g.root, "Player 1", ["Quit", "Pull"])
# If Player 1 quits immediately, he gets 0 and Player 2 wins.
g.set_outcome(g.root.children[0], p1_quits)

# Player 1 pulls with k = 6 -> chance: Fire with prob 1/6, NoFire with prob 5/6.
g.append_move(g.root.children[1], g.players.chance, ["Fire", "NoFire"])
g.set_chance_probs(g.root.children[1].infoset, [gbt.Rational(1, 6), gbt.Rational(5, 6)])
# If it fires, Player 1 shot himself.
g.set_outcome(g.root.children[1].children[0], p1_shot)
