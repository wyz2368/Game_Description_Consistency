import pygambit as gbt

# Reasoning / thought process in comments:
# - This is a sequential, common-knowledge game: after every trigger pull we know whether the gun fired
#   and how many unpulled chambers remain. Thus each decision node is fully distinguishable by
#   (remaining_chambers, current_player). No information sets are needed.
# - We unroll the entire game without loops or recursion. Start with 6 unpulled chambers, Player 1 to move.
# - At any decision node the player has two actions: "Quit" (guaranteed survival, payoff 0 to quitter, 1 to other)
#   or "Pull". If remaining_chambers > 1 then "Pull" leads to a chance node: "Fire" with probability 1/remaining,
#   "No fire" with probability (remaining-1)/remaining. If "No fire", remaining_chambers decreases by 1 and
#   play passes to the other player. If remaining_chambers == 1, pulling fires for certain and causes shooter death.
# - Payoffs: shooter dies -> (-1, 1) if Player 1 dies, (1, -1) if Player 2 dies.
# - We create all terminal outcomes ahead of time and reuse them.

g = gbt.Game.new_tree(players=["Player 1", "Player 2"],
                      title="Six-chamber alternating Russian roulette")

# Pre-create outcome objects for reuse
p1_dies = g.add_outcome([-1, 1], label="P1 dies")
p2_dies = g.add_outcome([1, -1], label="P2 dies")
p1_quit = g.add_outcome([0, 1], label="P1 quits")
p2_quit = g.add_outcome([1, 0], label="P2 quits")

# ROOT: remaining_chambers = 6, Player 1 to move

g.append_move(g.root, "Player 1", ["Quit", "Pull"])

# Pull branch -> chance (6 chambers -> fire with prob 1/6)
g.append_move(g.root.children[1], g.players.chance, ["Fire", "No fire"])
g.set_chance_probs(g.root.children[1].infoset, [gbt.Rational(1, 6), gbt.Rational(5, 6)])
